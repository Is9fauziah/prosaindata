import requests
from bs4 import BeautifulSoup
import re

for i in range(1, 4):
    url = f"https://cookpad.com/id/cari/resep%20masakan%20indonesia?page={i}"
    response = requests.get(url)
    soup = BeautifulSoup(response.content, "html.parser")
    # Find the <ul> tag with class 'xs:mx-rg lg:mx-0 '
    results = soup.find('ul', {'class': 'xs:mx-rg lg:mx-0'})
    # Find all <li> tags with id starting with 'recipe_'
    resep_elements = results.find_all('li', {'id': re.compile('^recipe_\d+')})

    for resep_element in resep_elements:
        # Extract the recipe title from the <a> tag with class 'block-link__main'
        judul_element = resep_element.find("a", class_="block-link__main")
        # Extract the recipe ingredients from the <div> tag with class 'clamp-2 break-words'
        bahan_element = resep_element.find("div", class_="clamp-2 break-words")
        # Extract the recipe author from the <span> tag with class 'text-cookpad-gray-600 text-cookpad-12 md:text-cookpad-14'
        pembuat_element = resep_element.find("span", class_="text-cookpad-gray-600 text-cookpad-12 md:text-cookpad-14")
        # Extract the recipe image(s) from the <img> tag(s)
        image_element = resep_element.find_all('img')
        for image in image_element:
            # Print image source
            print(image['src'])
            # Print alternate text
            print(image['alt'])

import csv
import requests
import json
from bs4 import BeautifulSoup


data = []
for i in range(1, 172):
    url = f"https://pta.trunojoyo.ac.id/c_search/byprod/10/{i}"
    response = requests.get(url)
    soup = BeautifulSoup(response.content, "html.parser")
    results = soup.find(id="content")
    karya_ilmiah_elements = results.find_all(attrs={"data-id": True})

    for karya_ilmiah_element in karya_ilmiah_elements:
        judul_element = karya_ilmiah_element.find("a", class_="title")
        penulis_element = karya_ilmiah_element.find_all("div")[0].find("span").text
        dosen1_element = karya_ilmiah_element.find_all("div")[2].find("span").text
        dosen2_element = karya_ilmiah_element.find_all("div")[3].find("span").text
        abstrak_url = karya_ilmiah_element.find("a", class_="gray button").get("href")

        # request ke halaman abstrak
        abstrak_response = requests.get(abstrak_url)
        abstrak_soup = BeautifulSoup(abstrak_response.content, "html.parser")
        abstrak = abstrak_soup.find("p").text.strip()

        # store the data as dictionary
        item = {
            "judul": judul_element.text.strip(),
            "penulis": penulis_element,
            "dosen1": dosen1_element,
            "dosen2": dosen2_element,
            "abstrak_url": abstrak_url,
            "abstrak": abstrak
        }
        data.append(item)

# convert the data to JSON format and write to file
with open("hasil.json", "w") as file:
    json.dump(data, file, indent=4)

print("Data berhasil disimpan ke file hasil.json")
